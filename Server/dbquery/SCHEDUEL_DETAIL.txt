DROP TABLE IF EXISTS SCHEDUEL_DETAIL;
CREATE TABLE SCHEDULE_DETAIL
(
	SCHEDULEID INT,
	NTHDAY INT,
	SEQ INT,
	USEREID VARCHAR(255),
	REGIONID VARCHAR(50),
	REGIONNAME VARCHAR(2),
	PLACEID VARCHAR(255),
	PLACENAME  VARCHAR(255),
	PLACEADDRESS  VARCHAR(255),
	MAPX VARCHAR(100),
	MAPY VARCHAR(100),
	CREATEDTIME TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	STARTDAY TIMESTAMP,
	ENDDAY TIMESTAMP,
	PRIMARY KEY(SCHEDULEID, NTHDAY, SEQ)
);
